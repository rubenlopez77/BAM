export class ToolsTexts {
  public async cleanText(str: string): Promise<string> {
    if (!str) return '';

    return str
      // eslint-disable-next-line unicorn/prefer-string-replace-all
      .replace(/\s*\([^()]*\)/g, '') 
      .replaceAll('\n', ' ')           
      // eslint-disable-next-line unicorn/prefer-string-replace-all
      .replace(/\s+/g, ' ')         
      .trim();
  }
}
